SELECT count(1) as active_licenses, client_id from product_user_access_mapping puam inner join client_master cm on cm.id = puam.client_id and cm.is_active = 1 and cm.is_delete = 0  inner join user_master um on um.id = puam.user_id and um.type = 'PRODUCT' and um.is_active = 1 and um.is_delete = 0 where puam.is_active = 1 and puam.is_delete = 0 and puam.status = 'ACTIVE' group by client_id ;
